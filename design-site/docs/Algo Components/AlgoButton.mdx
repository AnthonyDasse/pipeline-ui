# AlgoButton

A button to connect to MyAlgo. 

Generates a popup window to connect to MyAlgo. Uppon success, returns first wallet address to the context and returnTo specified.

import BrowserOnly from '@docusaurus/BrowserOnly';

export const BrowserOnlyButton = (props) => {
  return (
    <BrowserOnly>{() => {
      const AlgoButton = require('pipeline-ui').AlgoButton;
      const Pipeline = require('pipeline-ui').Pipeline;
      return <AlgoButton wallet={Pipeline.init()} />;
      }
    }
    </BrowserOnly>
    );
    };

<BrowserOnlyButton />

## Use Example  

```jsx
import React, { Component } from 'react';
import { AlgoButton, Pipeline} from 'pipeline-ui';


class TestButton extends Component {

  constructor(props) {
    super(props);
    this.state = {
      Algaddress: ""
    }
  }
  
  myAlgoWallet = Pipeline.init();

  render() {
    return <div>
      <AlgoButton 
            wallet={this.myAlgoWallet} 
            context={this} 
            returnTo={"Algaddress"} 
            />
      <h1>{this.state.Algaddress}</h1>
    </div>
  }
}

export default TestButton;
```
